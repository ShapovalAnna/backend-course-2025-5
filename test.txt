# PUT – запис картинки
curl -X PUT --data-binary "@cat.jpg" http://localhost:8080/200

# GET – отримання картинки
curl -X GET http://localhost:8080/200 --output result.jpg

# DELETE – видалення картинки
curl -X DELETE http://localhost:8080/200

# GET після видалення – перевірка 404
curl -X GET http://localhost:8080/200

# POST – перевірка 405
curl -X POST http://localhost:8080/200
